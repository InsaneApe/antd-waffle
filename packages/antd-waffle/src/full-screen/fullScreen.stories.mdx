```tsx

import React from 'react';
import { FullScreen, useFullScreenHandle } from './fullScreen.component';
import { Button } from 'antd';

const FullScreenExample = () => {
  const handle = useFullScreenHandle();
  return (
    <>
      <Button
        onClick={() => {
          if (handle.active) {
            handle.exit();
          } else {
            handle.enter();
          }
        }}
      >
        放大
      </Button>
      <FullScreen handle={handle}>
        <div
          style={{ color: handle.active ? 'white' : 'black', fontSize: '48px' }}
        >
          我是需要放大全屏的文字
        </div>
      </FullScreen>
    </>
  );
};

export default FullScreenExample;

```

|   Name    |              Description              |                       Type                       |       Default       |
| :-------: | :-----------------------------------: | :----------------------------------------------: | :-----------------: |
|  handle   | 绑定 useFullScreenHandle 可使用 hooks |                 FullScreenHandle                 | useFullScreenHandle |
| onChange  |           点击全屏发生事件            | (state:boolean, handle:FullScreenHandle) => void |         ' '         |
| className |             添加全屏类名              |                      string                      |         ' '         |
